<?php
declare(strict_types = 1);

namespace Pluginkollektiv\AntispamBee\Entity;

class Hook {

	public const TYPE_ACTION = 'action';
	public const TYPE_FILTER = 'filter';

	private $name;
	private $type;
	private $priority;
	private $expected_arguments;

	public function __construct( string $name, string $type, int $priority = 10, int $expected_arguments = 1 ) {

		$this->name               = $name;
		$this->type               = $type;
		$this->priority           = $priority;
		$this->expected_arguments = $expected_arguments;
	}

	public function name() {
		return $this->name;
	}

	public function type() {
		return $this->type;
	}

	public function priority() {
		return $this->priority;
	}

	public function expected_arguments() {
		return $this->expected_arguments;
	}

	public static function from_name( string $name ) {
		return new self(
			$name,
			self::TYPE_ACTION
		);
	}
}
